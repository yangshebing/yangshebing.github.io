C void sort(int a[],int n) { int i, j, index; for(i = 0; i < n - 1; i++) {

    index = i;

    for(j = i + 1; j < n; j++) {
        if(a[index] > a[j])
        {
            index = j;
        }
    }

    if(index != i) {
        int temp = a[i];
        a[i] = a[index];
        a[index] = temp;
    }
}

}

11、 求两个整数的最大公约数。

int gcd(int a, int b) { int temp = 0; if (a < b) { temp = a; a = b; b = temp; } while (b != 0) { temp = a % b; a = b; b = temp; } return a;}
